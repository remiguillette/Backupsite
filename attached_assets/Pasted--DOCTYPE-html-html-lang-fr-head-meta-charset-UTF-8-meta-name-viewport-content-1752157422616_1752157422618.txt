    <style>
        /* --- Styles de base pour la page --- */
        body {
            /* Centre le contenu verticalement et horizontalement */
            display: flex;
            justify-content: center;
            align-items: center;
            min-height: 100vh;
            margin: 0;
            /* Couleur de fond sombre pour faire ressortir le texte */
            background-color: #1a1a1a;
            font-family: 'Montserrat', sans-serif;
            overflow: hidden;
        }

        /* --- Styles pour le texte animé --- */
        .waving-text {
            /* Taille de la police, responsive */
            font-size: clamp(2rem, 10vw, 6rem);
            font-weight: 900;
            text-transform: uppercase;
            text-align: center;
            line-height: 1.1;

            /* --- La magie de l'animation --- */

            /* 1. Création du dégradé de fond (les couleurs du drapeau) */
            /* Le dégradé est beaucoup plus large que le texte pour permettre l'effet de vague */
            background-image: linear-gradient(
                90deg, 
                #00693C 45%, /* Vert franco-ontarien */
                #FFFFFF 50%, /* Blanc */
                #00693C 55%  /* Vert franco-ontarien */
            );
            background-size: 400% 100%; /* Taille du dégradé (4x plus large) */

            /* 2. Le texte est utilisé comme un "masque" pour le dégradé */
            background-clip: text;
            -webkit-background-clip: text; /* Compatibilité pour Safari/Chrome */

            /* 3. Le texte lui-même est rendu transparent pour laisser voir le dégradé en arrière-plan */
            color: transparent;
            /* La propriété standard `color` ne fonctionne pas toujours avec `background-clip`.
               On utilise `-webkit-text-fill-color` pour un meilleur contrôle. */
            -webkit-text-fill-color: transparent;

            /* 4. Application des deux animations en parallèle */
            /* - `wave-effect`: Fait bouger le dégradé pour l'effet de vague. */
            /* - `color-cycle`: Gère le cycle entre la couleur orange et l'effet de drapeau. */
            animation: 
                wave-effect 3s linear infinite,
                color-cycle 12s linear infinite;
        }

        /* --- Définition des animations (@keyframes) --- */

        /* Animation 1: L'effet de vague */
        /* Fait défiler la position du dégradé de droite à gauche */
        @keyframes wave-effect {
            0% {
                background-position: 100% 50%;
            }
            100% {
                background-position: 0% 50%;
            }
        }

        /* Animation 2: Le cycle des couleurs */
        /* Contrôle la visibilité du dégradé ou de la couleur orange unie */
        @keyframes color-cycle {
            /* Début du cycle: le texte est orange pendant 40% de la durée */
            0%, 40% {
                /* On force la couleur du texte à être orange, ce qui cache le dégradé */
                -webkit-text-fill-color: #ff6b35;
            }
            /* Milieu du cycle: l'effet de drapeau est visible pendant 40% de la durée */
            50%, 90% {
                /* On rend le texte transparent, ce qui révèle le dégradé qui bouge en arrière-plan */
                -webkit-text-fill-color: transparent;
            }
            /* Fin du cycle: retour à l'orange */
            100% {
                -webkit-text-fill-color: #ff6b35;
            }
        }
